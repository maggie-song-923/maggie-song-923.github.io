let lang = "zh";
let screen = "intro";
let qIndex = 0;
let answers = [];
let result = null;

const $ = id => document.getElementById(id);
const L = () => DATA[lang];

const TONE_COLOR = {"温柔":"var(--rose)","奇趣":"var(--teal)","坚韧":"var(--clay)"};
const TONE_SOFT  = {"温柔":"#C58B8120","奇趣":"#5C8C8F20","坚韧":"#BE8B4520"};
function cn(n){ return ["零","一","二","三","四","五","六","七","八","九","十"][n]||n; }

function setScreen(s){
  screen = s;
  ["intro","quiz","result"].forEach(x=>$(x).classList.toggle("on", x===s));
  window.scrollTo(0,0);
}

function render(){
  $("toggle").textContent = L().ui.toggle;
  $("credit").textContent = L().ui.credit;
  if(screen==="intro") renderIntro();
  else if(screen==="quiz") renderQ(qIndex);
  else renderResult();
}

function renderIntro(){
  const m=L().meta, u=L().ui;
  $("i-eyebrow").textContent = u.eyebrow;
  $("i-title").textContent = m.title;
  $("i-sub").textContent = "— " + m.subtitle;
  $("i-lede").textContent = m.intro;
  $("i-note").textContent = m.note;
  $("i-start-t").textContent = u.start;
}

function renderQ(i){
  qIndex = i;
  setScreen("quiz");
  const q = L().questions[i], total = L().questions.length, u = L().ui;
  const isLast = (i === total-1);
  $("q-counter").textContent = (i+1) + " / " + total;
  $("q-bar").style.width = (i/total*100) + "%";
  $("q-num").textContent = isLast ? u.qLast : (lang==="zh" ? ("第"+cn(i+1)+"题") : (u.qWord + (i+1)));
  $("q-text").textContent = q.q;
  $("q-back").textContent = u.back;
  $("q-back").classList.toggle("show", i>0);
  const box = $("q-opts"); box.innerHTML = "";
  q.options.forEach((o,idx)=>{
    const b = document.createElement("button");
    b.className = "opt"; b.textContent = o.t;
    b.addEventListener("click", ()=>{
      answers[i] = idx;
      if(i+1 < total) renderQ(i+1);
      else { $("q-bar").style.width = "100%"; finish(); }
    });
    box.appendChild(b);
  });
}

function finish(){
  const d = {"社交":{"群":0,"独":0},"风险":{"探":0,"守":0},"节奏":{"冲":0,"续":0}};
  let tone = "温柔";
  answers.forEach((idx,i)=>{
    const dim = DATA.zh.questions[i].dim;
    const p = DATA.zh.questions[i].options[idx].p;
    if(dim==="底色") tone = p; else d[dim][p]++;
  });
  const social = d["社交"]["群"] >= 2 ? "群" : "独";
  const risk   = d["风险"]["探"] >= 2 ? "探" : "守";
  const tempo  = d["节奏"]["冲"] >= 2 ? "冲" : "续";
  result = { key: social+risk+tempo, tone, social, risk, tempo };
  renderResult();
}

function renderResult(){
  setScreen("result");
  const u = L().ui, ax = L().axis;
  const r = L().results[result.key][result.tone];
  document.documentElement.style.setProperty("--tone", TONE_COLOR[result.tone]);
  document.documentElement.style.setProperty("--tone-soft", TONE_SOFT[result.tone]);

  const box = $("r-axes"); box.innerHTML = "";
  [ax.social[result.social], ax.risk[result.risk], ax.tempo[result.tempo], ax.tone[result.tone]].forEach(t=>{
    const s = document.createElement("span"); s.className = "tag"; s.textContent = t; box.appendChild(s);
  });

  $("r-pre").textContent = u.rPre;
  $("r-name").textContent = r.animal;
  $("r-mirror").textContent = r.animal;
  $("r-habit-label").textContent = u.habitLabel;
  $("r-habit").textContent = r.habit;
  $("r-intro").textContent = r.intro;
  $("lbl-strength").textContent = u.secStrength;
  $("lbl-shadow").textContent = u.secShadow;
  $("lbl-niche").textContent = u.secNiche;
  $("lbl-reframe").textContent = u.secReframe;
  $("r-strength").textContent = r.strength;
  $("r-shadow").textContent = r.shadow;
  $("r-niche").textContent = r.niche;
  $("r-reframe").textContent = r.reframe;
  $("r-again").textContent = u.again;
  $("r-share").textContent = u.share;
}

$("toggle").addEventListener("click", ()=>{ lang = (lang==="zh" ? "en" : "zh"); render(); });
$("i-start").addEventListener("click", ()=>{ answers=[]; result=null; renderQ(0); });
$("q-back").addEventListener("click", ()=>{ if(qIndex>0) renderQ(qIndex-1); });
$("r-again").addEventListener("click", ()=>{ answers=[]; result=null; setScreen("intro"); renderIntro(); });

render();
